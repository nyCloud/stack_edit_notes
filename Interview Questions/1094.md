from collections import defaultdict  
  
  
def car_pooling(trips, capacity):  
    stops, events = [], defaultdict(int)  
    for t in trips:  
        stops += t[1:]  
        events[t[1]] += t[0]  
        events[t[2]] -= t[0]  
      
    cnt = 0  
  stops = sorted(list(set(stops)))  
    for s in stops:  
        cnt += events[s]  
        if s > capacity:
